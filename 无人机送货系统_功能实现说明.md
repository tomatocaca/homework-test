```cpp
int readInt() {
    int value;

        cin.clear();
        cin.ignore(numeric_limits<streamsize>::max(), '\n');
        cout << "输入无效，请输入数字: ";
    }
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    return value;
}

double readDouble() {
    double value;
    while (!(cin >> value)) {
        cin.clear();
        cin.ignore(numeric_limits<streamsize>::max(), '\n');

}
```

**功能：** 防止用户输入非数字字符导致 `cin` 进入 fail 状态、循环无限刷屏。检测到非法输入时自动清除错误状态、丢弃残余字符，并提示重新输入。

```cpp
    for (auto& d : mediumDrones) if (d.getDroneState() != Idle) busy++;
```

### 4.2 Drone 基类及子类

```cpp
class Drone {
protected:
    string droneID;          // 编号 L001/M001/H001
    DroneState droneState;   // 当前状态
    DroneLevel level;        // 无人机等级
    double maxWeight;        // 最大载重
    int deliveryCount;       // 当日配送次数
    string assignedOrderID;  // 当前分配的订单编号（"#"=无）
    string assignedBy;       // 分配此订单的配送员编号
    string currentLocation;  // 当前位置（站点代号）
    time_t deliveryStartTime;// 配送开始时间戳
    int deliveryTotalMinutes;// 预计总时长
    string targetStation;    // 配送目标站点

public:
    Drone();  // 默认构造

    virtual string getSpecialFeatures() const = 0;      // 纯虚：特殊功能描述
    virtual void printSpecialActions(const Order&) const {} // 虚函数钩子：子类专属提示
    virtual void deliver(vector<Order>& allOrders);     // 虚函数：执行配送

    // Getter
    string getDroneID() const;
    int getDeliveryCount() const;
    DroneState getDroneState() const;
    DroneLevel getLevel() const;
    double getMaxWeight() const;
    string getAssignedOrderID() const;
    string getAssignedBy() const;
    string getCurrentLocation() const;
    time_t getDeliveryStartTime() const;
    int getDeliveryTotalMinutes() const;
    int getRemainingMinutes() const;  // 计算配送剩余时长

    // Setter
    void setAssignedOrderID(string oid);
    void setAssignedBy(string ab);

    bool hasOrder() const;  // 检查是否有分配的订单
    virtual void Show();    // 控制台输出无人机信息
};
```

```cpp
Account* login(const string& role) {
```

### 6.2 userRegister() — 用户注册

```cpp
void userRegister() {
    clearScreen();
    cout << "\n=========== 用户注册 ===========" << endl;

    string username, password, confirmPwd, idCard, phone;

    cout << "用户名: "; getline(cin, username);
    for (size_t i = 0; i < accounts.size(); i++) {
        if (accounts[i].username == username) {
            cout << "该用户名已被注册！注册失败。" << endl;
            waitForReturn(); return;
        }
    }

    cout << "密码: "; getline(cin, password);
    cout << "确认密码: "; getline(cin, confirmPwd);
    if (password != confirmPwd) {
        cout << "两次密码不一致！注册失败。" << endl;
        waitForReturn(); return;
    }

    cout << "身份证号: "; getline(cin, idCard);
    if (idCard.empty()) {
        cout << "身份证号不能为空！注册失败。" << endl;
        waitForReturn(); return;
    }

    cout << "电话号码: "; getline(cin, phone);
    if (phone.empty()) {
        cout << "电话号码不能为空！注册失败。" << endl;
        waitForReturn(); return;
    }

    Account newAcc;
    newAcc.username = username;
    newAcc.password = password;
    newAcc.role = "user";
    newAcc.displayName = username;
    newAcc.idCard = idCard;
    newAcc.phone = phone;
    newAcc.cancelCount = 0;
    newAcc.cancelWeekStart = 0;
    newAcc.bannedUntil = 0;
    accounts.push_back(newAcc);

    cout << "\n注册成功！请使用用户名 \"" << username << "\" 登录。" << endl;
    waitForReturn();
}
```

**功能：** 收集用户名、密码（两次确认）、身份证号、电话号码，检查用户名是否重复，创建新 Account 加入全局列表。

## 7. 管理员功能

### 7.1 adminAddDrone() — 添加无人机

**代码实现：**
```cpp
void adminAddDrone() {
    clearScreen();
    cout << "\n=========== 添加无人机 ===========" << endl;
    cout << "请选择无人机等级：" << endl;
    cout << "  1. 轻型（载重≤" << LIGHT_MAX_WEIGHT << "kg）" << endl;
    cout << "  2. 中型（载重≤" << MEDIUM_MAX_WEIGHT << "kg）" << endl;
    cout << "  3. 重型（载重≤" << HEAVY_MAX_WEIGHT << "kg）" << endl;
    cout << "请选择 (1-3): ";
    int levelChoice = readInt();

    string prefix;
    switch (levelChoice) {
        case 1: prefix = "L"; break;
        case 2: prefix = "M"; break;
        case 3: prefix = "H"; break;
        default: cout << "无效选择！" << endl; return;
    }

    cout << "请输入无人机编号（" << prefix << "+3位数字）: ";
    string newID; cin >> newID;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');

    bool duplicate = false;
    if (levelChoice == 1) {
        for (auto& d : lightDrones) if (d.getDroneID() == newID) duplicate = true;
        if (!duplicate) { lightDrones.push_back(LightDrone(newID, Idle));
            cout << "轻型无人机 " << newID << " 添加成功！" << endl; }
    } else if (levelChoice == 2) {
        for (auto& d : mediumDrones) if (d.getDroneID() == newID) duplicate = true;
        if (!duplicate) { mediumDrones.push_back(MediumDrone(newID, Idle));
            cout << "中型无人机 " << newID << " 添加成功！" << endl; }
    } else if (levelChoice == 3) {
        for (auto& d : heavyDrones) if (d.getDroneID() == newID) duplicate = true;
        if (!duplicate) { heavyDrones.push_back(HeavyDrone(newID, Idle));
            cout << "重型无人机 " << newID << " 添加成功！" << endl; }
    }
    if (duplicate) cout << "错误：无人机 " << newID << " 已存在！" << endl;
    waitForReturn();
}
```

**功能：** 选择等级 → 输入编号 → 检查重复 → 创建对应类型无人机加入列表。

```cpp
            && selectedOrder.getWeight() <= mediumDrones[i].getMaxWeight()
            && mediumDrones[i].getDeliveryCount() < MAX_DAILY_DELIVERIES)
            options.push_back({1, (int)i});
    }
    for (size_t i = 0; i < heavyDrones.size(); i++) {
        if (heavyDrones[i].getDroneState() == Idle
            && heavyDrones[i].canCarry(selectedOrder.getCargoType())
            && selectedOrder.getWeight() <= heavyDrones[i].getMaxWeight()
            && heavyDrones[i].getDeliveryCount() < MAX_DAILY_DELIVERIES)
            options.push_back({2, (int)i});

            features = mediumDrones[options[i].droneIndex].getSpecialFeatures();

            delCnt = heavyDrones[options[i].droneIndex].getDeliveryCount();
```

```cpp
    vector<int> cancelIndices;
    for (size_t i = 0; i < allOrders.size(); i++) {
        if (allOrders[i].getOwnerName() == userName
            && (allOrders[i].getOrderState() == Pending
                || allOrders[i].getOrderState() == Assigned)) {
            cout << ++cancelCount << ". 订单 " << allOrders[i].getOrderID()
                 << " | " << cargoTypeToString(allOrders[i].getCargoType())
                 << " | 状态：" << orderStateToString(allOrders[i].getOrderState()) << endl;
            cancelIndices.push_back(i);
        }
    }

    if (cancelCount == 0) {
        cout << "您没有可取消的订单！" << endl; waitForReturn(); return;
    }

    cout << "请选择要取消的订单 (1-" << cancelCount << "): ";
    int choice = readInt();
    if (choice < 1 || choice > cancelCount) {
        cout << "无效选择！" << endl; waitForReturn(); return;
    }

    int orderIndex = cancelIndices[choice - 1];
    cout << "\n确认取消订单 " << allOrders[orderIndex].getOrderID() << "？(y/n): ";
    char confirm; cin >> confirm;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    if (confirm != 'y' && confirm != 'Y') {
        cout << "已取消操作。" << endl; waitForReturn(); return;
    }

    if (allOrders[orderIndex].getOrderState() == Assigned) {
        releaseOrderDrone(allOrders[orderIndex].getOrderID());
    }

    allOrders[orderIndex].setOrderState(Cancelled);
    cout << "\n订单 " << allOrders[orderIndex].getOrderID() << " 已取消！" << endl;
    recordCancellation(*myAcc);
    waitForReturn();
}
```

## 10. 封禁机制

### 10.1 isBanned() — 检查封禁状态

```cpp
bool isBanned(const Account& acc) {
    if (acc.bannedUntil == 0) return false;
    time_t now = time(0);
    return now < acc.bannedUntil;
}
```

### 10.2 resetCancelCountIfNewWeek() — 跨周重置

```cpp
void resetCancelCountIfNewWeek(Account& acc) {
    time_t now = time(0);
    tm* nowTm = localtime(&now);
    tm weekStartTm = *nowTm;
    weekStartTm.tm_hour = 0; weekStartTm.tm_min = 0; weekStartTm.tm_sec = 0;
    int daysFromMonday = nowTm->tm_wday == 0 ? 6 : nowTm->tm_wday - 1;
    weekStartTm.tm_mday -= daysFromMonday;
    time_t thisMonday = mktime(&weekStartTm);
    if (acc.cancelWeekStart < thisMonday) {
        acc.cancelCount = 0;
        acc.cancelWeekStart = thisMonday;
    }
}
```

### 10.3 recordCancellation() — 记录取消并触发封禁

```cpp
void recordCancellation(Account& acc) {
    resetCancelCountIfNewWeek(acc);
    acc.cancelCount++;
    cout << "（本周已取消 " << acc.cancelCount << " 次）" << endl;
    if (acc.cancelCount >= 3) {
        time_t now = time(0);
        acc.bannedUntil = now + 14 * 24 * 60 * 60;
        cout << "⚠ 您的本周取消次数已达3次，账号已被封禁至 "
             << formatBanTime(acc.bannedUntil) << "！" << endl;
    }
}
```

## 12. 输入健壮性与防刷屏

### 问题背景

`cin >> int_var` 读取非数字字符时，`cin` 进入 fail 状态，错误字符残留缓冲区，循环无法退出导致无限刷屏。

### 解决方案

封装 `readInt()` / `readDouble()`，检测到非法输入时自动 `cin.clear()` + `cin.ignore()` 并重试。

### 替换范围

共 **17 处** `cin >> int/double` 全部替换，覆盖所有菜单选择和数值输入。

## 13. 代码优化说明

### 13.1 性能优化：字符串传引用

所有函数参数中的 `string` 类型从值传递改为 `const string&` 引用传递，避免不必要的字符串拷贝：

```cpp
// 优化前（值传递，每次调用拷贝一次字符串）
void deliverAssignOrder(string delivererID);
void userPlaceOrder(string userName);

// 优化后（引用传递，零拷贝）
void deliverAssignOrder(const string& delivererID);
void userPlaceOrder(const string& userName);
```

涉及函数：`displayDeliverMenu`、`deliverAssignOrder`、`deliverExecute`、`deliverShowRecords`、`deliverLoop`、`displayUserMenu`、`userPlaceOrder`、`userShowMyOrders`、`userConfirmReceive`、`userCancelOrder`、`userLoop`，以及 `Order` 构造函数、`Drone` 构造函数和 setter。

### 13.2 结构优化：提取查找/统计辅助函数

将多处重复的查找和统计逻辑提取为独立工具函数：

| 函数 | 替代的重复代码 | 调用次数 |
|------|--------------|---------|
| `findAccountByName()` | 手动遍历 accounts 查找用户名 | 3处 |
| `findOrderByID()` | 手动遍历 allOrders 查找订单编号 | 1处 |
| `releaseOrderDrone()` | 取消订单时释放无人机的20行代码 | 1处 |
| `countBusyDrones()` | 统计忙碌无人机的3个循环 | 1处 |

### 13.3 结构优化：虚函数钩子消除子类重复代码

三个子类（LightDrone/MediumDrone/HeavyDrone）的 `deliver()` 方法原本各有 40+ 行几乎相同的代码。通过引入 `printSpecialActions()` 虚函数钩子，将差异部分（温控/保险锁提示）抽离：

```cpp
// 基类：通用配送逻辑 + 调用钩子
class Drone {
    virtual void printSpecialActions(const Order&) const {}  // 默认空实现
    virtual void deliver(vector<Order>& allOrders);          // 通用实现
};

// 子类：只需重写钩子
class MediumDrone : public Drone {
    void printSpecialActions(const Order& order) const override {
        if (order.getCargoType() == Fresh || order.getCargoType() == Medicine)
            cout << "【温控系统已启动】..." << endl;
    }
    void deliver(vector<Order>& allOrders) override { Drone::deliver(allOrders); }
};
```

**效果：** 删除约 120 行重复代码，三个子类的 `deliver()` 从各 40+ 行缩减为 1 行调用。

### 13.4 可读性优化：消除魔法数字

将代码中硬编码的数字提取为命名常量：

| 魔法数字 | 常量名 | 含义 |
|---------|--------|------|
| `3` | `CANCEL_THRESHOLD` | 每周取消次数封禁阈值 |
| `14` | `BAN_DAYS` | 封禁天数 |
| `4` | `STATION_COUNT` | 站点总数 |

### 13.5 可读性优化：range-for 替代索引循环

将 `for (size_t i = 0; i < vec.size(); i++) vec[i]...` 模式统一替换为 `for (auto& item : vec) item...`，代码更简洁。
